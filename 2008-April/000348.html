 package com.flagstone.transform;
 
+/**
+ * The Displayable interface is implemented by all classes that are visible on
+ * the Flash Player screen.
+ * 
+ * The principal use of the interface is to allow the width and height of objects 
+ * representing images and shapes to obtained without having to test the class
+ * or type and cast to the correct class.
+ */
 public interface Displayable
 {
+	/**
+	 * Return the width of an object in twips. The value returned does not take
+	 * into account any scaling or other coordinate transform that might be 
+	 * performed to resize the object when it is rendered on the screen.
+	 * 
+	 * @return the width of the object in twips.
+	 */
 	public int getWidth();
+
+	/**
+	 * Return the height of an object in twips. The value returned does not take
+	 * into account any scaling or other coordinate transform that might be 
+	 * performed to resize the object when it is rendered on the screen.
+	 * 
+	 * @return the height of the object in twips.
+	 */
 	public int getHeight();
 }
