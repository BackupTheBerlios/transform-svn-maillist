+package com.flagstone.transform;
+
+/**
+ * FSLineStyle defines the width and colour of a line that is used when drawing
+ * the outline of a shape.
+ * 
+ * <table class="datasheet">
+ * 
+ * <tr>
+ * <th align="left" colspan="2">Attributes</th>
+ * </tr>
+ * 
+ * <tr>
+ * <td>width</td>
+ * <td>The width of the line drawn in twips.</td>
+ * </tr>
+ * 
+ * <tr>
+ * <td>color</td>
+ * <td>The colour used to draw the line.</td>
+ * </tr>
+ * 
+ * </table>
+ * 
+ * <p>
+ * All lines are drawn with rounded corners and end caps. Different join and
+ * line end styles can be created by drawing line segments as a sequence of
+ * filled shapes. With 1 twip equal to 1/20th of a pixel this technique can
+ * easily be used to draw the narrowest of visible lines.
+ * </p>
+ * 
+ * <p>
+ * Whether the alpha channel in the colour is used is determined by the class
+ * used to define the shape. Transparent colours are only supported from Flash 3
+ * onwards. Simply specifying the level of transparency in the FSColor object is
+ * not sufficient.
+ * </p>
+ * 
+ * <h1 class="datasheet">Examples</h1>
+ * 
+ * <pre>
+ * FSLineStyle lineStyle = new FSLineStyle(20, new FSColor(0, 0, 0));
+ * </pre>
+ * 
+ * <p>
+ * Draws a black line, 1 pixel wide.
+ * </p>
+ * 
+ * <pre>
+ * FSLineStyle lineStyle = new FSLineStyle(20, new FSColor(255, 0, 0, 128));
+ * </pre>
+ * 
+ * <p>
+ * Draws a transparent red line, 1 pixel wide. If the alpha channel is not
+ * specified then the value defaults to 255 generating a completely opaque
+ * colour.
+ * </p>
+ * 
+ * <p>
+ * Flash only supports contiguous lines. Dashed line styles can be created by
+ * drawing the line as a series of short line segments by interspersing
+ * FSShapeStyle objects to move the current point in between the FSLine objects
+ * that draw the line segments.
+ * </p>
+ * 
+ * <pre>
+ * // define the line style
+ * lineStyles.add(new FSLineStyle(20, new FSColor(0, 0, 0)));
+ * 
+ * // Select the line style
+ * shape.add(new FSShapeStyle(1, 0, 0));
+ * 
+ * // Draw the dashed line with lines and gaps each 5 pixels long
+ * shape.add(new FSShapeStyle(1, 0, 0));
+ * 
+ * for (int i = 0; i < 10; i++)
+ * {
+ * 	shape.add(new FSLine(100, 0));
+ * 	shape.add(new FSShapeStyle(0, 0, 0));
+ * 	shape.add(new FSLine(100, 0));
+ * }
+ * </pre>
+ * 
+ * <h1 class="datasheet">History</h1>
+ * 
+ * <p>
+ * The FSLineStyle class represents the LineStyle structure from the Macromedia
+ * Flash (SWF) File Format Specification. It was introduced in Flash 1. Support
+ * for transparent colours was added in Flash 3.
+ * </p>
+ */
+public class FSLineStyle implements Cloneable, Codeable
+{
+	private int width;
+	private FSColor color;
+
+	/**
+	 * Construct an FSLineStyle object, initalizing it with values decoded from
+	 * an encoded object.
+	 * 
+	 * @param coder
+	 *            an FSCoder containing the binary data.
+	 */
+	public FSLineStyle(FSCoder coder)
+	{
+		decode(coder);
+	}
+
+	/**
+	 * Constructs an FSLineStyle, specifying the width and colour of the line.
+	 * 
+	 * @param aWidth
+	 *            the width of the line.
+	 * @param aColor
+	 *            the colour of the line.
+	 */
+	public FSLineStyle(int aWidth, FSColor aColor)
+	{
+		setWidth(aWidth);
+		setColor(aColor);
+	}
+
+	/**
+	 * Constructs an FSLineStyle object by copying values from an existing
+	 * object.
+	 * 
+	 * @param obj
+	 *            an FSLineStyle object.
+	 */
+	public FSLineStyle(FSLineStyle obj)
+	{
+		width = obj.width;
+		color = new FSColor(obj.color);
+	}
+
+	/**
+	 * Gets the width of the line.
+	 * 
+	 * @return the width of the line.
+	 */
+	public int getWidth()
+	{
+		return width;
+	}
+
+	/**
+	 * Gets the colour of the line.
+	 * 
+	 * @return the colour for the line.
+	 */
+	public FSColor getColor()
+	{
+		return color;
+	}
+
+	/**
+	 * Sets the width of the line.
+	 * 
+	 * @param aNumber
+	 *            the width of the line.
+	 *            
+	 * @throws IllegalArgumentException if the argument is not in the range 0..65535.
+	 */
+	public void setWidth(int aNumber)
+	{
+		if (aNumber < 0 || aNumber > 65535) {
+			throw new IllegalArgumentException("Line width must be in range 0..65535.");
+		}
+		width = aNumber;
+	}
+
+	/**
+	 * Sets the colour of the line.
+	 * 
+	 * @param aColor
+	 *            the colour of the line.
+	 * 
+	 * @throws NullPointerException if the argument is null,
+	 */
+	public void setColor(FSColor aColor)
+	{
+		if (aColor == null) {
+			throw new NullPointerException("Colour cannot be null");
+		}
+		color = aColor;
+	}
+
+	public Object clone()
+	{
+		FSLineStyle anObject = null;
+
+		try
+		{
+			anObject = (FSLineStyle) super.clone();
+			anObject.color = (FSColor) color.clone();
+		} 
+		catch (CloneNotSupportedException e)
+		{
+			throw new InternalError();
+		}
+		return anObject;
+	}
+
+	public boolean equals(Object anObject)
+	{
+		boolean result = false;
+
+		if (super.equals(anObject))
+		{
+			FSLineStyle typedObject = (FSLineStyle) anObject;
+
+			result = width == typedObject.width;
+			result = result && color.equals(typedObject.color);
+		}
+		return result;
+	}
+
+	public String toString()
+	{
+		StringBuffer buffer = new StringBuffer();
+
+		buffer.append("FSLineStyle : {");
+		buffer.append("width = ").append(width).append("; ");
+		buffer.append("color = ").append(color.toString()).append("; ");
+		buffer.append("}");
+
+		return buffer.toString();
+	}
+
+	public int length(FSCoder coder)
+	{
+		int length = 2;
+
+		length += color.length(coder);
+
+		return length;
+	}
+
+	public void encode(FSCoder coder)
+	{
+		coder.writeWord(width, 2);
+		color.encode(coder);
+	}
+
+	public void decode(FSCoder coder)
+	{
+		width = coder.readWord(2, false);
+		color = new FSColor(coder);
+	}
+}
